22 MagickPPExport int Magick::operator == (const Magick::Color &left_,
23  const Magick::Color &right_)
24 {
25 #if defined(MAGICKCORE_HDRI_SUPPORT)
26  return((left_.isValid() == right_.isValid()) &&
27  (fabs(left_.quantumRed()-right_.quantumRed()) < MagickEpsilon) &&
28  (fabs(left_.quantumGreen()-right_.quantumGreen()) < MagickEpsilon) &&
29  (fabs(left_.quantumBlue()-right_.quantumBlue()) < MagickEpsilon));
30 #else
31  return((left_.isValid() == right_.isValid()) &&
32  (left_.quantumRed() == right_.quantumRed()) &&
33  (left_.quantumGreen() == right_.quantumGreen()) &&
34  (left_.quantumBlue() == right_.quantumBlue()));
35 #endif
36 }

253 Magick::Color::operator std::string() const
254 {
255  char
256  colorbuf[MagickPathExtent];
257 
258  PixelInfo
259  pixel;
260 
261  if (!isValid())
262  return std::string("none");
263 
264  pixel.colorspace=(_pixelType == RGBPixel || _pixelType == RGBAPixel) ?
265  sRGBColorspace : CMYKColorspace;
266  pixel.depth=MAGICKCORE_QUANTUM_DEPTH;
267  pixel.alpha=_pixel->alpha;
268  pixel.alpha_trait=_pixel->alpha_trait;
269  pixel.black=_pixel->black;
270  pixel.blue=_pixel->blue;
271  pixel.green=_pixel->green;
272  pixel.red=_pixel->red;
273  GetColorTuple(&pixel,MagickTrue,colorbuf);
274 
275  return(std::string(colorbuf));
276 }

421 void Magick::Color::setAlpha(const Magick::Quantum alpha_)
422 {
423  _pixel->alpha=alpha_;
424  if (alpha_ == OpaqueAlpha)
425  {
426  _pixel->alpha_trait=UndefinedPixelTrait;
427  if (_pixelType == RGBAPixel)
428  _pixelType=RGBPixel;
429  else if (_pixelType == CMYKAPixel)
430  _pixelType=CMYKPixel;
431  }
432  else
433  {
434  _pixel->alpha_trait=BlendPixelTrait;
435  if (_pixelType == RGBPixel)
436  _pixelType=RGBAPixel;
437  else if (_pixelType == CMYKPixel)
438  _pixelType=CMYKAPixel;
439  }
440 }
